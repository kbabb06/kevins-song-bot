<!DOCTYPE html>

<html>

<body style="background:#111;color:#fff;font-family:sans-serif;max-width:600px;margin:40px auto;">


<h1>Request a Song</h1>


<input id="search" placeholder="Search..." style="width:100%;padding:10px;" />

<button onclick="searchSongs()" style="width:100%;padding:10px;margin-top:10px;">Search</button>


<div id="results"></div>


<script>

async function searchSongs() {

  const q = document.getElementById("search").value;


  const res = await fetch('/api/songs?q=' + encodeURIComponent(q));

  const data = await res.json();


  const container = document.getElementById("results");

  container.innerHTML = "";


  data.forEach(song => {

    const div = document.createElement("div");

    div.style.padding = "10px";

    div.style.borderBottom = "1px solid #333";

    div.innerText = song.song + " - " + song.artist;


    div.onclick = async () => {

      await fetch('/api/queue', {

        method: 'POST',

        headers: { 'Content-Type': 'application/json' },

        body: JSON.stringify(song)

      });


      alert("Queued: " + song.song);

    };


    container.appendChild(div);

  });

}

</script>


</body>

</html>
